42 /* For TCPC the registers are not mapped to local memory, we need to use an intermediate buffer uint8_t msg_buf[32]; */
43 /* When sending TRANSMIT_BUFFER through I2c, the TRANSMIT_BYTE_COUNT and TX_BUF_HEADER_BYTE_0/1 must been sent together
44  */
45 /* with TX_BUF_OBJn_BYTE as per USB-Port Conroller Specification 4.3.5 Writing the TRANSMIT_BUFFER */
46 /* So define this array to contain all buf content. */
47 /* When transmiting, 0-reserved, 1-TRANSMIT_BYTE_COUNT, 2-TX_BUF_HEADER_BYTE_0, 3-TX_BUF_HEADER_BYTE_1, */
48 /* 4~31-TX_BUF_OBJn_BYTE */
49 /* When receiving, 0-RECEIVE_BYTE_COUNT, 1-RX_BUF_FRAME_TYPE, 2-RX_BUF_HEADER_BYTE_0, 3-RX_BUF_HEADER_BYTE_1, */
50 /* 4~31-RX_BUF_OBJn_BYTE */
