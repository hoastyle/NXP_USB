242 /*
243  * The following MACROs (USB_GLOBAL, USB_BDT, USB_RAM_ADDRESS_ALIGNMENT, etc) are only used for USB device stack.
244  * The USB device global variables are put into the section m_usb_global and m_usb_bdt or the section
245  * .bss.m_usb_global and .bss.m_usb_bdt by using the MACRO USB_GLOBAL and USB_BDT. In this way, the USB device
246  * global variables can be linked into USB dedicated RAM by USB_STACK_USE_DEDICATED_RAM.
247  * The MACRO USB_STACK_USE_DEDICATED_RAM is used to decide the USB stack uses dedicated RAM or not. The value of
248  * the marco can be set as 0, USB_STACK_DEDICATED_RAM_TYPE_BDT_GLOBAL, or USB_STACK_DEDICATED_RAM_TYPE_BDT.
249  * The MACRO USB_STACK_DEDICATED_RAM_TYPE_BDT_GLOBAL means USB device global variables, including USB_BDT and
250  * USB_GLOBAL, are put into the USB dedicated RAM. This feature can only be enabled when the USB dedicated RAM
251  * is not less than 2K Bytes.
252  * The MACRO USB_STACK_DEDICATED_RAM_TYPE_BDT means USB device global variables, only including USB_BDT, are put
253  * into the USB dedicated RAM, the USB_GLOBAL will be put into .bss section. This feature is used for some SOCs,
254  * the USB dedicated RAM size is not more than 512 Bytes.
255  */
256 #define USB_STACK_DEDICATED_RAM_TYPE_BDT_GLOBAL 1
257 #define USB_STACK_DEDICATED_RAM_TYPE_BDT 2
